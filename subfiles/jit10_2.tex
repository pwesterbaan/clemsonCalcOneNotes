    \section{JIT 10.2: Special Formulas}
      \begin{enumerate}
        \item $x^2-y^2=(x+y)\cdot(x-y)$
        \item $x^3+y^3=(x+y)(x^2-xy+y^2)$
        \item $x^3-y^3=(x-y)(x^2+xy+y^2)$
        \item $x^2+2xy+y^2=(x+y)^2$
        \item $x^2-2xy+y^2=(x-y)^2$
        \item $acx^2+(bc+ad)x+bd=(ax+b)(cx+d)$
      \end{enumerate}
      \begin{ex*}
        Factor: 
        \begin{itemize}
          \item $z^2-9$\\
          \item $x^4-y^4$\\
          \item $(x-y)^2-4y^2$\\
          \item $a^3+8b^3$\\
          \item $27x^3+64y^3z^6$\\
          \item $x^2+5x+6$
        \end{itemize}
      \vspace*{\stretch{1}}
      \end{ex*}
