  \textbf{Theorem 5.1: Sums of Powers of Integers}
  
  Let $n$ be a positive integer and $c$ a real number.
  \begin{align*}
    \sum\kto^n c&=cn& 
    \sum\kto^n k &=\frac{n(n+1)}{2}\\
    \sum\kto^n k^2&=\frac{n(n+1)(2n+1)}{6}&
    \sum\kto^n k^3&=\frac{n^2(n+1)^2}{4}
  \end{align*}
}}
